CREATE TABLE ASSIGNMENTS (
	ID	INTEGER	NOT NULL	GENERATED ALWAYS
	AS IDENTITY (START WITH 1, INCREMENT BY 1, CACHE 20,
	   NO MINVALUE, NO MAXVALUE, NO CYCLE, NO ORDER),
	EMPLOYEE_ID	VARCHAR(20),
	ROLE	VARCHAR(20),
	PROJECT_ID	INTEGER,
	START_DATE	DATE,
	END_DATE	DATE,
	STATUS	VARCHAR(20)
);

CREATE TABLE CLIENTS (
	ID	INTEGER	NOT NULL	GENERATED ALWAYS
	AS IDENTITY (START WITH 1, INCREMENT BY 1, CACHE 20,
	   NO MINVALUE, NO MAXVALUE, NO CYCLE, NO ORDER),
	NAME	VARCHAR(50),
	RELATIONSHIP	VARCHAR(50),
	CP_FIRST_NAME	VARCHAR(50),
	CP_LAST_NAME	VARCHAR(50),
	CP_TITLE	VARCHAR(50),
	CP_PREFERRED_NAME	VARCHAR(50),
	CP_EMAIL	VARCHAR(50),
	CP_STREET	VARCHAR(50),
	CP_NUMBER	VARCHAR(50),
	CP_CITY	VARCHAR(50),
	CP_STATE	VARCHAR(50),
	CP_POST_CODE	VARCHAR(50),
	CP_COUNTRY	VARCHAR(50),
	STATUS	VARCHAR(20),
	CP_MOBIE_PHONE	VARCHAR(50),
	CP_DESK_PHONE	VARCHAR(50)
);

	   NO MINVALUE, NO MAXVALUE, NO CYCLE, NO ORDER),

CREATE TABLE EMPLOYEES (

	   NO MINVALUE, NO MAXVALUE, NO CYCLE, NO ORDER),

CREATE TABLE LINE_ITEMS (
	ID	INTEGER	NOT NULL	GENERATED ALWAYS
		AS IDENTITY (START WITH 1, INCREMENT BY 1, CACHE 20,
		NO MINVALUE, NO MAXVALUE, NO CYCLE, NO ORDER),
	RATE	VARCHAR(20),
	NAME	VARCHAR(100),

	TYPE_ID	INTEGER,
	CURRENCY_ID	INTEGER
);

CREATE TABLE PROJECTS (
	ID	INTEGER	NOT NULL	GENERATED ALWAYS
	AS IDENTITY (START WITH 1, INCREMENT BY 1, CACHE 20,
	   NO MINVALUE, NO MAXVALUE, NO CYCLE, NO ORDER),
	CODE	VARCHAR(20),
	NAME	VARCHAR(100),
	STATUS	VARCHAR(20),
	DESCRIPTION	VARCHAR(500),
	START_DATE	DATE,
	END_DATE	DATE,
	CLIENT_ID	INTEGER,
	PROJECT_MANAGER	VARCHAR(100)
);

CREATE TABLE PROJECTS_LINEITEMS (

);

CREATE TABLE TIMESHEETS (
